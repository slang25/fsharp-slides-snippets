type PaymentMethod = 
  | Cash
  | Cheque of int
  | Card of CardType * CardNumber

let printPayment method = 
  match method with
  | Cash ->
      printfn "Paid in cash"
  | Cheque checkNo ->
      printfn "Paid by cheque: %i" checkNo
  | Card (cardType,cardNo) ->
      printfn "Paid with %A %A" cardType cardNo
